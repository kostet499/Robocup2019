34  {
35  // header buffer
36  ConstBuffer<N> headerBuffer = N::buffer(static_cast<const void*>(&msg.header()),
37  sizeof(Message::Header));
38  std::vector<ConstBuffer<N>> buffers;
39  const auto& msgBuffer = msg.buffer();
40 
41  // A buffer has a header and data.
42  // Inside data, subbuffers' sizes are interleaved.
43  // Subbuffers are _not_ inside the buffer, but in separate memory.
44  // We're going to send to the network:
45  // - the header
46  // - for all subbuffers:
47  // - the data chunk in the main buffer up to (and including) the subbuffer's size
48  // - the subbuffer
49  // - the last data chunk in the main buffer
50  //
51  // Memory layout for a buffer with 2 subbuffers:
52  // (low address) (high address)
53  // |header|buffer_part_0|size_subbuffer_0|buffer_part_1|size_subbuffer_1|buffer_part_2|
54  buffers.reserve(1 + 2 * msgBuffer.subBuffers().size() + 1);
55  buffers.push_back(headerBuffer);
56 
57  decltype(msgBuffer.size()) beginOffset = 0;
58  // subbuffers
59  for (const auto& sub: msgBuffer.subBuffers())
60  {
61  // buffer chunk between startOffset and the offset past the next subbuffer's size
62  const auto sizeOffset = sub.first;
63  auto endOffset = sizeOffset + sizeof(Buffer::size_type);
64  if (endOffset != beginOffset)
65  buffers.push_back(N::buffer(
66  static_cast<const char*>(msgBuffer.data()) + beginOffset, endOffset - beginOffset));
67  beginOffset = endOffset;
68  // subbuffer
69  const auto& subBuffer = sub.second;
70  buffers.push_back(N::buffer(subBuffer.data(), subBuffer.size()));
71  }
72  // end of main buffer
73  buffers.push_back(N::buffer(
74  static_cast<const char*>(msgBuffer.data()) + beginOffset, msgBuffer.size() - beginOffset));
75  return buffers;
76  }
